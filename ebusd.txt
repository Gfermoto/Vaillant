# ebusd configuration for HASSio addon (LukasGrebe/ha-addons)
# Формат YAML ниже подходит для дополнения eBUSd до ~25.1.
#
# С дополнения 26.1+ схема другая: commandline_options, каталог /addon_configs/…_ebusd/,
# локальный --configpath к …/en. См. HA_ADDON_EBUSD_26.md и ebusd-addon-26.example.txt
#
# For Docker usage, see docker-compose.yml

# Auto-pick CSV config files matching the initial scan result
scanconfig: full

# Log level: use "notice" for normal operation, "debug" only for troubleshooting
loglevel_all: notice

# MQTT topic prefix
mqtttopic: ebusd

# MQTT integration config for Home Assistant discovery
mqttint: /config/ebusd/mqtt-hassio.cfg

# Publish in JSON format (required for mqtt-hassio.cfg integration)
mqttjson: true

# Full scan to detect all devices on the bus
scan: full

# eBUS adapter address — change to your adapter IP:port
# Prefix "ens:" = enhanced high-speed protocol (recommended for ebusd adapter v5)
network_device: 192.168.1.142:9999
mode: ens

# Extra transfer latency in ms (increase if you see many CRC errors)
latency: 10

# Retry counts for improved reliability
acquireretries: 10
receivetimeout: 200
sendretries: 5
scanretries: 10

# Allow all access levels (required for write commands)
accesslevel: "*"
